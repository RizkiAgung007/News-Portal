table likes
CREATE TABLE likes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_users INT NOT NULL,
  id_news VARCHAR(500) NOT NULL,
  value TINYINT(1) NOT NULL, -- 1 untuk like, 0 untuk dislike
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  CONSTRAINT fk_user FOREIGN KEY (id_users) REFERENCES users(id_users)
);
